    ```javascript
      /**
       * GET /api/daily/suggestions
       * Returns { suggestions, rerolls_remaining, reset_time }
       * Triggers LLM generation if not cached; returns cache otherwise.
       */
      getSuggestions: () => api.get('/api/daily/suggestions').then(r => r.data),

      accept: (index) => api.post(`/api/daily/accept/${index}`).then(r => r.data),

      /**

       * 429 if rerolls exhausted.
       */
      reroll: (index) => api.post(`/api/daily/reroll/${index}`).then(r => r.data),
    };
    ```

    **Handler functions (add alongside existing handlers):**
    ```javascript
    const handleGetDailySuggestions = async () => {
      setIsDailyLoading(true);
      try {
        const data = await dailyService.getSuggestions();
        setDailySuggestions(data.suggestions);
        setDailyRerollsRemaining(data.rerolls_remaining);
        setDailyResetTime(data.reset_time);
      } catch (err) {
        setToast({ message: 'Ошибка получения квестов дня', type: 'error' });
      } finally {
        setIsDailyLoading(false);
      }
    };

    const handleAcceptSuggestion = async (index) => {
      setAcceptingIndex(index);
      try {
        await dailyService.accept(index);
        // Re-fetch suggestions to get updated list from backend
        const data = await dailyService.getSuggestions();
        setDailySuggestions(data.suggestions);
        setDailyRerollsRemaining(data.rerolls_remaining);
        // Invalidate active quests query to show new quest
        queryClient.invalidateQueries({ queryKey: ['quests'] });
        setToast({ message: 'Квест добавлен!', type: 'success' });
        if (typeof triggerHaptic === 'function') triggerHaptic();
      } catch (err) {
        if (err.response?.status === 409) {
          setToast({ message: 'Освободи слот (5/5 активных)', type: 'error' });
        } else {
          setToast({ message: 'Ошибка принятия квеста', type: 'error' });
        }
      } finally {
        setAcceptingIndex(null);
      }
    };

    const handleRerollSuggestion = async (index) => {
      if (dailyRerollsRemaining <= 0) return;
      setRerollingIndex(index);
      try {
        const data = await dailyService.reroll(index);
        setDailySuggestions(data.suggestions);
        setDailyRerollsRemaining(data.rerolls_remaining);
      } catch (err) {
        if (err.response?.status === 429) {
          setToast({ message: 'Лимит рероллов исчерпан (2/2)', type: 'error' });
        } else {
          setToast({ message: 'Ошибка реролла', type: 'error' });
        }
      } finally {
        setRerollingIndex(null);
      }
    };
    ```
